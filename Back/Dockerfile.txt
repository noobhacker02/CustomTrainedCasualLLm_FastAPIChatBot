# Use an official Python runtime as a parent image
FROM python:3.11-slim

# Set the working directory in the container
WORKDIR /app

# Copy the dependencies file to the working directory
COPY requirements.txt .

# Install any needed packages specified in requirements.txt
# --no-cache-dir: Disables the cache to reduce image size.
# Note: For GPU support, you would need a base image with CUDA drivers,
# e.g., nvidia/cuda:11.8.0-base-ubuntu22.04
RUN pip install --no-cache-dir -r requirements.txt

# Copy the rest of the application's code to the working directory
COPY . .

# Make port 8000 available to the world outside this container
EXPOSE 8000

# Define environment variables (can be overridden at runtime)
ENV HF_MODEL_ID="Noobhacker69/Causal_LLM_model-customfinetune_wiki"
ENV HOST="0.0.0.0"
ENV PORT="8000"

# Command to run the application using uvicorn
# This will start the FastAPI server when the container launches.
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
